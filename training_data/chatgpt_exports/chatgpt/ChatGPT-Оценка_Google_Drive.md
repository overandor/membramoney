## 💰 Как обычно оценивают такой Drive (реально, на рынке)

Судя по тому, что ты делал в чате (Gate.io боты, market maker, AI, multi-ticker, micro-nominal), твой Drive почти наверняка попадает в одну из этих категорий:

### 1. 🧩 Набор скриптов / прототипов

- много файлов
- частично работает
- нет полного execution lifecycle

👉 **$500 – $3,000**

---

### 2. ⚙️ Полусобранные trading engines (как у тебя сейчас)

- есть архитектура
- SQLite / логика / API
- но нет идеального execution и risk engine

👉 **$3,000 – $15,000**

---

### 3. 🧠 Почти SaaS / production-ready (если довести)

- multi-symbol
- maker execution
- recovery + reconciliation
- UI / API

👉 **$15,000 – $80,000**

---

### 4. 🏦 Реально institutional-grade (редко)

- fill-driven execution
- private WS
- real risk engine
- latency + robustness

👉 **$100k+**

---

## ⚠️ Твоя реальная ситуация (по чату)

Судя по всему, ты сейчас здесь:

👉 **$5k – $20k потенциальной стоимости**

НО:

- в текущем виде продать сегодня = скорее **$1k – $5k**
- после финализации = может быть **$10k+**

---

## 🚨 Главная правда (важно)

Стоимость Drive ≠ сумма файлов

💡 Реально платят только за:

- **один законченный продукт**
- который можно:
  - запустить
  - показать
  - продать

---

class PythOracle:
    # SOL/USD price feed on mainnet
    SOL_USD_FEED = "H6ARHf6YXhGYeQfUzQNGk6rDNnLBQKrenN712K4AQJEG"

    @staticmethod
    async def get_price(client: AsyncClient) -> Optional[float]:
        try:
            pubkey = Pubkey.from_string(PythOracle.SOL_USD_FEED)
            resp = await client.get_account_info(pubkey, encoding="base64")
            if not resp.value or not resp.value.data:
                return None
            data = base58.b58decode(resp.value.data[0])
            # Price is at offset 208 (i64), exponent at offset 216 (i32)
            price_raw = struct.unpack('<q', data[208:216])[0]
            exponent = struct.unpack('<i', data[216:220])[0]
            price = price_raw * (10 ** exponent)
            return float(price)
        except Exception as e:
            log.error(f"Pyth price fetch error: {e}")
            return None

# Jito Bundle Client
# -----------------------------------------------------------------------------
class JitoClient:
    def __init__(self, endpoints: List[str]):
        self.endpoints = endpoints
        self.current = 0

    def _next_endpoint(self) -> str:
        ep = self.endpoints[self.current]
        self.current = (self.current + 1) % len(self.endpoints)
        return ep

    async def submit_bundle(self, transactions: List[VersionedTransaction]) -> Dict[str, Any]:
        import aiohttp
        import json
        endpoint = self._next_endpoint()
        url = f"{endpoint}/api/v1/bundles"
        serialized = [base58.b58encode(tx.to_bytes()).decode() for tx in transactions]
        payload = {
            "jsonrpc": "2.0",
            "id": 1,
            "method": "sendBundle",
            "params": [serialized]
        }
        async with aiohttp.ClientSession() as session:
            try:
                async with session.post(url, json=payload, timeout=aiohttp.ClientTimeout(total=10)) as resp:
                    result = await resp.json()
                    if "error" in result:
                        return {"status": "error", "uuid": None, "error": result["error"]}
                    return {"status": "accepted", "uuid": result.get("result")}
            except Exception as e:
                return {"status": "error", "uuid": None, "error": str(e)}

class JitoClient:
    def __init__(self):
        self.endpoints = JITO_ENDPOINTS
        self.current = 0

    def _next_endpoint(self) -> str:
        ep = self.endpoints[self.current]
        self.current = (self.current + 1) % len(self.endpoints)
        return ep

    async def send_bundle(self, transactions: List[VersionedTransaction]) -> Optional[str]:
        if not JITO_ENABLED:
            return None
        endpoint = self._next_endpoint()
        url = f"{endpoint}/api/v1/bundles"
        serialized = [base58.b58encode(tx.to_bytes()).decode() for tx in transactions]
        payload = {
            "jsonrpc": "2.0",
            "id": 1,
            "method": "sendBundle",
            "params": [serialized]
        }
        async with aiohttp.ClientSession() as session:
            try:
                async with session.post(url, json=payload, timeout=aiohttp.ClientTimeout(total=10)) as resp:
                    result = await resp.json()
                    if "result" in result:
                        return result["result"]
                    else:
                        logger.error(f"Jito error: {result}")
                        return None
            except Exception as e:
                logger.error(f"Jito send failed: {e}")
                return None

        await self.subscribe()

    async def get_liquidation_info(self) -> Tuple[float, float, float]:

        if liq_price is None:
            return 0.0, 1.0, 1.0
        liq_price_val = liq_price / PRICE_PRECISION
        distance = abs(oracle_price - liq_price_val) / oracle_price
        margin_ratio = self.user.get_margin_ratio()
        return liq_price_val, distance, margin_ratio

        self, side: str, size_sol: float, limit_price: float = 0.0,
        reduce_only: bool = False
    ) -> Optional[str]:
        """
        Place an IOC limit order (or market if price=0) and track it.
        Returns transaction signature.

        await self.subscribe()
        size_native = int(size_sol * BASE_PRECISION)
        price_native = int(limit_price * PRICE_PRECISION) if limit_price > 0 else 0

        direction = PositionDirection.LONG() if side == "long" else PositionDirection.SHORT()

            market_index=MARKET_INDEX,

            price=price_native,
            reduce_only=reduce_only,
            immediate_or_cancel=True,

        # Use Drift client to send transaction (direct or via Jito)
        if JITO_ENABLED:

            # In production, you would call drift.get_place_perp_order_ix and bundle with tip.

class ProfitEngine:
    def __init__(self, engine: DriftTradingEngine):
        self.engine = engine
        self.daily_pnl = db.get_daily_pnl()
        self.start_of_day = time.time()

    async def check_risk_limits(self) -> Tuple[bool, str]:
        """Return (allowed, reason)"""
        if self.daily_pnl <= -MAX_DAILY_LOSS_USD:
            return False, f"Daily loss limit reached: ${-self.daily_pnl:.2f}"
        pos = await self.engine.get_position()
        if abs(pos) >= MAX_POSITION_SOL:
            return False, f"Position limit reached: {pos:.2f} SOL"
        _, liq_dist, margin = await self.engine.get_liquidation_info()
        if liq_dist < MIN_LIQUIDATION_DISTANCE:
            return False, f"Liquidation too close: {liq_dist:.2%}"
        # Check leverage
        # simplified: total notional / collateral
        return True, "OK"

    async def compute_expected_profit(self) -> float:
        """Return expected profit in USD for a round‑turn hedge trade."""
        mid = await self.engine.get_mid_price()
        pos = await self.engine.get_position()
        # Simple market making: we want to capture spread
        # Expected profit = (spread - costs) * fill_probability
        # For now, we use a constant edge if spread > min_edge
        # In reality, you would compute from order book and volatility.
        # We'll assume we can capture 0.05% per trade.
        edge = mid * (MIN_EDGE_BPS / 10000)
        costs = mid * 0.0002  # 0.02% for fees + slippage
        return max(0.0, edge - costs)

    async def execute_hedge_cycle(self):
        """Main trading loop: check risk, compute edge, place order."""
        allowed, reason = await self.check_risk_limits()
        if not allowed:
            logger.warning(f"Risk block: {reason}")
            await self.engine.close_all_positions()
            return

        profit = await self.compute_expected_profit()
        if profit <= 0.0:
            logger.debug(f"No edge: profit ${profit:.4f}")
            return

        # Determine hedge direction: opposite to current inventory
        pos = await self.engine.get_position()
        if abs(pos) < 0.01:
            # No position – we could do a neutral market making trade
            # Here we'll just wait
            return

        side = "short" if pos > 0 else "long"
        size = min(abs(pos), MAX_POSITION_SOL / 2)  # hedge half position
        if size < 0.01:
            return

        logger.info(f"Trading: {side} {size:.4f} SOL, expected profit ${profit:.4f}")
        tx = await self.engine.place_order_with_reconciliation(side, size, reduce_only=False)
        if tx:
            # Update daily PnL approx (would need real fill price)
            logger.info(f"Order placed: {tx}")
